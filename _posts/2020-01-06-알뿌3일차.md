데이터 묶어서 정렬하기 : 객체로 정리
```go
#include <iostream>
#include <algorithm>

using namespace std;

class Student{
    public:
        string  name;
        int     score;
        Sudent (string name, int score){
            this -> name = name;
            this -> score = score;
        }
        bool operator < (Student &student){
            return this -> score < student.score;
        }
}

int main(void){
    Student students[] = {
        Student("AAA", 90);
        Student("BBB", 90);
        Student("CCC", 90);
        Student("DDD", 90);
        Student("FFF", 90);
    };
    sort(students, students + 5);
}

```
